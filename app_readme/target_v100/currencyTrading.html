1、获取交易区和交易货币类别

说明：

接口负责人：刘富国
请求方式：POST
请求示例： 
{
    "ver": "1.0.0",
    "token": "xxx",
    "server": "GetTradeArea"
     "data": {
    "first_entrust_info":"1"  //只显示一个交易货币类别：0，显示全部，1：只显示第一个
    }
}
成功返回： 
{
    "data": [
         {
            "area_name": "BTC", //BTC 交易区
            "area_id": "1",     //交易区ID
            "child_info": [
                {
                    "entrust_type": "1",  //交易货币类别ID
                    "coin_str": "LTC/BTC" //交易货币类别名称
                    "entrust_type": "1",    //委托类型ID
                    "coin_str": "LTC/BTC",  //委托类型名称
                    "high": "0.01959128",   //最高价
                    "low": "0.01661228",    //最低价
                    "vol": "519731.93921170", //成交量(最近的24小时)
                    "rtq": "0.01917554",    //最后成交价
                    "p_currency_id": "1",     //主货币类型
                    "p_currency_name": "BTC",     //主货币名称
                    "currency_id": "2",     //目标货币类型
                    "currency_name": "LTC",     //目标货币类型
                    "rate": "0.17%",        //每日涨跌比例
                    "perc_status": "1"      //涨跌标识  1 涨 -1跌
                    "sell": "0.01863345",   //卖出价
                    "sell_fee": "0.0010",   //卖出手续费
                    "buy": "0.01863344",    //买入价
                    "buy_fee": "0.0010"     //买入手续费
                    "float_price_rate": "0.1000"  //限价单：委托价格价格浮动上限,如果是0或者是空，则不限制
                }
            ]
        },
    ],
    "error": "0",
    "msg": "success"
}
失败返回：（有多种情况） 
{
    "data": {},
    "error": "错误码",
    "msg": "错误说明"
}

4、订单查询

说明：

接口负责人：刘富国
请求方式：POST
请求示例： 
{
    "ver": "1.0.0",
    "token": "xxxx",
    "data": {
        "area_id":1,    //交易区ID
        "type":1,       //1表示正在交易,2表示撤销/已完成订单
        "entrust_type":2    //委托类别 (可选)
        "page": "1",  //页数
        "limit": "10"  // 每页显示行数
    },
    "server": "GetCurrencyTradingOrderList"
}
成功返回： 
{
    "data": {
        "total": "1",
        "list": [
             {
                "order_id": "66",           //订单ID
                "entrust_type": "4",        //委托类别
                "coin": "BCH/BTC",          //委托类别名称,
                "transaction_type": "2",    //1为买，2为卖
                "entrust_num": "1.00000000",  //委托数量
                "entrust_price": "0.11550000", //委托单价
                "entrust_money": "0.11550000", //委托总额： 委托数量*委托单价
                "avg_price": "0.1155",         //成交均价,
                "success_num": "1.00000000",   //成交数量
                "sum_money": "0.11550000"      //成交总额
                "status_str": "已完成",        //订单状态
                "update_time": "1530093882",   //撤销时间或完成时间
                "add_time": "1530093682"
            }
        ],
        "pager": {
            "current_page": "1",
            "last_page": "",
            "next_page": "",
            "total_pages": "1"
        }
    },
    "error": "0",
    "msg": "success"
}
失败返回：（有多种情况） 
{
    "data": {},
    "error": "错误码",
    "msg": "错误说明"

5、买单/卖单

说明：

接口负责人：刘富国
请求方式：POST
请求示例： 
{
    "ver": "1.0.0",
    "token": "xxxx",
    "data": {
        "area_id":1,    //交易区ID
        "price_type":2,    // 价格类型，1表示限价单，2表示市价单
        "entrust_type":2 ,   //委托类别
        "transaction_type": "2", // 交易类型，1表示买入，2表示卖出
        "entrust_price": "0.01417069", //委托单价
        "total_price": "50", //选择市价单时买入/卖出总数量
        "leave_num": "50",  //选择限价时候的买入/卖出数量
        "trade_pwd": "xxx", //资金密码
    },
    "server": "ProcessTradeInfo"
}
成功返回： 
                                最后一条未完成的订单信息
                              {
    "data": {
        "order_id": "181",  //订单ID
        "entrust_type": "1",  //委托类别
        "coin": "LTC/BTC"  ,//委托类别名称
        "entrust_num": "10.00000000", //委托数量
        "entrust_price": "0.01811191",//单价价格
        "entrust_money": "0.18111910", //总价（参考价）
        "success_num": "0.00000000", // 成交数量
        "leave_num": "10.00000000",  //未成交数量
        "add_time": "1514367264", //添加时间
        "transaction_type": "2" //1为买，2为卖
    },
    "error": "0",
    "msg": "success"
}
                                或者：
                                匹配完毕
{
    "data": {
       "is_success": "1"
    },
    "error": "0",
    "msg": "success"
}
失败返回：（有多种情况） 
{
    "data": {},
    "error": "错误码",
    "msg": "错误说明"
}
