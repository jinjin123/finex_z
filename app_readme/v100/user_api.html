1、发送单条极光消息

说明：

接口负责人：刘富国
请求方式：POST
请求示例： 
{
    "server": "SendMsgToPerson",
    "data": {
        "uid": "49",  //用户ID
        "title": "标题"  // 消息标题
        "content": "内容"  // 消息内容
         //拓展字段，根据业务设置
         "extras": "{
                send_modle":"C2C"，//C2C，P2P订单模块
                "new_order_penging":"1"   //有新订单要处理
                }
    }
}
成功返回： 
{
    "data": {
    "is_success": "1"
  },
    "error": "0",
    "msg": "success"
}
失败返回：（有多种情况） 
{
    "data": {},
    "error": "10025",
    "msg": "消息发送失败"
}

2、批量发送极光消息

说明：

接口负责人：刘富国
请求方式：POST
请求示例： 
{
    "server": "SendMsgToPersonList",
    "data": {
        "send_msg_list":[ {
                        "uid": "49",  //用户ID
                        "title": "标题"  // 消息标题
                        "content": "内容"  // 消息内容
                          //拓展字段，根据业务设置
                         "extras": "{
                                send_modle":"C2C"，//C2C，P2P订单模块
                                "new_order_penging":"1"   //有新订单要处理
                                }

                        } ]
    }
}
成功返回： 
{
    "data": {
    "error_list": "0"
  },
    "error": "0",
    "msg": "success"
}

                                或者：

{
"data": {
"error_list": "[{
            "uid":"181",//用户ID
            "errmsg":"无设备ID号",//备注
                }]"
},
"error": "0",
"msg": "success"
}
失败返回：（有多种情况） 
{
    "data": {},
    "error": "10025",
    "msg": "消息发送失败"
}
